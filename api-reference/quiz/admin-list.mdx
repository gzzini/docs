---

## Descrição

Lista todos os quizzes criados no sistema, ordenados por data de criação (mais recente primeiro). Este endpoint é útil para administração e monitoramento do sistema de quizzes.

<Warning>
  Este é um endpoint administrativo que retorna todos os quizzes do sistema.
</Warning>

## Response

<ResponseField name="success" type="boolean" required>
  Indica se a operação foi bem-sucedida
</ResponseField>

<ResponseField name="data" type="object" required>
  <Expandable title="Dados da resposta">
    <ResponseField name="total" type="number" required>
      Total de quizzes no sistema
    </ResponseField>

    <ResponseField name="quizzes" type="array" required>
      Lista de todos os quizzes
      <Expandable title="Estrutura de cada quiz">
        <ResponseField name="id" type="string">
          ID único do quiz
        </ResponseField>
        <ResponseField name="title" type="string">
          Título do quiz
        </ResponseField>
        <ResponseField name="description" type="string">
          Descrição do quiz
        </ResponseField>
        <ResponseField name="quizDate" type="string">
          Data do quiz (formato YYYY-MM-DD)
        </ResponseField>
        <ResponseField name="postId" type="string">
          ID do post do Beehiiv relacionado
        </ResponseField>
        <ResponseField name="minCorrectAnswers" type="number">
          Mínimo de acertos para passar
        </ResponseField>
        <ResponseField name="createdAt" type="string">
          Data/hora de criação (formato ISO 8601)
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
